Tool Setup Configs

Claude CLI / Desktop

// ~/.claude.json or claude_desktop_config.json
{
  "mcpServers": {
    "memory": {
      "command": "npx",
      "args": ["@universalmemory/mcp-server"],
      "env": {
        "MEMORY_API_KEY": "mem_sk_xxx"
      }
    }
  }
}

VS Code / Copilot

// .vscode/mcp.json
{
  "servers": {
    "memory": {
      "type": "stdio",
      "command": "npx",
      "args": ["@universalmemory/mcp-server"],
      "env": {
        "MEMORY_API_KEY": "${input:apiKey}"
      }
    }
  }
}
